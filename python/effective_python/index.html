用生成器表达式来改写数据量较大的列表推导

value = [len(x) for x in open('test.txt')]
print(value) # [100, 57, 15, 1, 11]
it = (len(x) for x in open('test.txt'))
print(it) # <generator object <genexpr> at 0x101b81480>
print(next(it)) # 100
print(next(it)) # 57
roots = ((x, x**0.5) for x in it)
